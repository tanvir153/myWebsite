import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime
import numpy as np

# 1) Load & inspect data
file_path = "my_data.json"  
print(f"Loading data from {file_path} …")
df = pd.read_json(file_path)  
print("Columns in dataframe:", df.columns.tolist())

# 2) Parse dates, sort, and basic cleaning
df['date'] = pd.to_datetime(df['date'], errors='coerce')
df = df.dropna(subset=['date', 'value']).sort_values('date').reset_index(drop=True)
print("\nHead of cleaned data:")
display(df.head())
print("\nSummary statistics of 'value' (Sales):")
display(df['value'].describe())

# 3) Time‐Series Line Chart
plt.figure(figsize=(10, 5))
plt.plot(df['date'], df['value'], marker='o', linestyle='-', alpha=0.8)
plt.title("Amazon Sales Trend Over Time", fontsize=16)
plt.xlabel("Date")
plt.ylabel("Sales Value")
plt.grid(alpha=0.3)
plt.tight_layout()
plt.savefig("amazon_sales_trend_over_time.png")
plt.show()

# 4) 7-Day Rolling Average
df['rolling_7d'] = df['value'].rolling(window=7, min_periods=1).mean()
plt.figure(figsize=(10, 5))
plt.plot(df['date'], df['rolling_7d'], color='orange', linewidth=2)
plt.title("7-Day Rolling Average of Amazon Sales", fontsize=16)
plt.xlabel("Date")
plt.ylabel("Sales (7-Day Avg)")
plt.grid(alpha=0.3)
plt.tight_layout()
plt.savefig("amazon_sales_7d_rolling_avg.png")
plt.show()

# 5) Distribution of Sales
plt.figure(figsize=(8, 5))
sns.histplot(df['value'], bins=20, kde=True, color='teal')
plt.title("Distribution of Amazon Sales Values", fontsize=16)
plt.xlabel("Sales Value")
plt.ylabel("Frequency")
plt.tight_layout()
plt.savefig("amazon_sales_distribution.png")
plt.show()

# 6) Monthly Sales Boxplot (Seasonality)
df['month'] = df['date'].dt.month
plt.figure(figsize=(10, 6))
sns.boxplot(x='month', y='value', data=df, palette='viridis')
plt.title("Monthly Amazon Sales Distribution", fontsize=16)
plt.xlabel("Month")
plt.ylabel("Sales Value")
plt.tight_layout()
plt.savefig("amazon_sales_by_month.png")
plt.show()

# 7) Sales vs. Day of Month Scatter
df['day'] = df['date'].dt.day
plt.figure(figsize=(8, 5))
plt.scatter(df['day'], df['value'], alpha=0.7, c=df['month'], cmap='plasma')
plt.title("Amazon Sales vs. Day of Month", fontsize=16)
plt.xlabel("Day of Month")
plt.ylabel("Sales Value")
cbar = plt.colorbar()
cbar.set_label("Month")
plt.tight_layout()
plt.savefig("amazon_sales_vs_day.png")
plt.show()

# 8) Heatmap: Avg Sales by Month & Day
pivot = df.pivot_table(
    index=df['date'].dt.day,
    columns=df['date'].dt.month,
    values='value',
    aggfunc='mean'
)
plt.figure(figsize=(10, 6))
sns.heatmap(pivot, cmap='coolwarm', linewidths=0.5)
plt.title("Heatmap of Avg Amazon Sales by Month & Day", fontsize=16)
plt.xlabel("Month")
plt.ylabel("Day of Month")
plt.tight_layout()
plt.savefig("amazon_sales_heatmap.png")
plt.show()

# 9) Save cleaned & enriched data
out_csv = "amazon_sales_cleaned.csv"
df.to_csv(out_csv, index=False)
print(f"\nCleaned sales data saved to {out_csv}")
